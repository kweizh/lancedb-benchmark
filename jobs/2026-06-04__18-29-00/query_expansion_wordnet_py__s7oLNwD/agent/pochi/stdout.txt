from nltk.corpus import wordnet as wn
token = 'car'
expanded = [token]
seen = {token}
for synset in wn.synsets(token):
    for lemma in synset.lemmas():
        name = lemma.name()
        if '_' in name or ' ' in name:
            continue
        if name not in seen:
            seen.add(name)
            expanded.append(name)
            if len(expanded) - 1 >= 3:
                break
    if len(expanded) - 1 >= 3:
        break
print('Expanded terms for car:', expanded)
