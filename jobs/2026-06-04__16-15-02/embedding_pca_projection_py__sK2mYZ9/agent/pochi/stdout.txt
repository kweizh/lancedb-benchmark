import lancedb
import numpy as np
from sklearn.decomposition import PCA
import os

# 1. Read all data from source table
db = lancedb.connect('/home/user/myproject/lancedb')
tbl = db.open_table('articles')
data = tbl.to_arrow()

ids = data.column('id').to_pylist()
titles = data.column('title').to_pylist()
embeddings = data.column('embedding').to_pylist()

print(f'Loaded {len(ids)} rows')
print(f'id type: {type(ids[0])}, first: {ids[0]}')
print(f'Embedding shape check: {len(embeddings[0])}')

# 2. Convert embeddings to numpy array
X = np.array(embeddings, dtype=np.float32)
print(f'X shape: {X.shape}')

# 3. Fit PCA deterministically (svd_solver='full' is deterministic)
pca = PCA(n_components=16, svd_solver='full', random_state=42)
X_pca = pca.fit_transform(X)

print(f'PCA components shape: {pca.components_.shape}')
print(f'PCA mean shape: {pca.mean_.shape}')
print(f'X_pca shape: {X_pca.shape}')

# 4. Save PCA model to /app/pca_model.npz
np.savez('/app/pca_model.npz', components=pca.components_, mean=pca.mean_)
print('Saved PCA model to /app/pca_model.npz')

# Verify the saved model
loaded = np.load('/app/pca_model.npz')
print(f'Saved components shape: {loaded[\"components\"].shape}')
print(f'Saved mean shape: {loaded[\"mean_\"].shape}')

# 5. Create new LanceDB table
run_id = os.environ['ZEALT_RUN_ID']
table_name = f'articles_pca_{run_id}'
print(f'Creating table: {table_name}')

# Prepare data for the new table
import pyarrow as pa

new_ids = ids  # keep same ids
new_titles = titles
new_embeddings = X_pca.astype(np.float32).tolist()  # 16-d vectors
new_original_ids = ids  # original_id = id from source

# Build pyarrow table
# embedding must be fixed-size list of 16 float32
embedding_type = pa.list_(pa.float32(), 16)

table_data = pa.table({
    'id': pa.array(new_ids, type=pa.int64()),
    'title': pa.array(new_titles, type=pa.string()),
    'embedding': pa.array(new_embeddings, type=embedding_type),
    'original_id': pa.array(new_original_ids, type=pa.int64()),
})

print(f'New table schema: {table_data.schema}')
print(f'New table row count: {len(table_data)}')

# Drop existing table if it exists and create new one
try:
    db.drop_table(table_name)
    print(f'Dropped existing table {table_name}')
except:
    pass

new_tbl = db.create_table(table_name, table_data)
