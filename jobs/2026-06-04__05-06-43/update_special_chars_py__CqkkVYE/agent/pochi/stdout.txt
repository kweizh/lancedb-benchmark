The Python `values=` form sidesteps this footgun entirely because the value is bound, not interpolated, into SQL. import os
import lancedb
import pyarrow as pa
import numpy as np
import json

uri = os.environ.get("LANCEDB_URI", "/workspace/db")
db = lancedb.connect(uri)

schema = pa.schema([
    pa.field("id", pa.int64()),
    pa.field("author", pa.string()),
    pa.field("body", pa.string()),
    pa.field("vector", pa.list_(pa.float32(), 4))
])

rng = np.random.default_rng(seed=42)

data = []
for i in range(1, 9):
    data.append({
        "id": i,
        "author": f"author_{i}",
        "body": f"body_{i}",
        "vector": rng.random(4).astype(np.float32).tolist()
    })

tbl = db.create_table("notes", data=data, schema=schema, mode="overwrite")

tbl.update(where="id = 2", values={"body": "I'm good"})
tbl.update(where="id = 4", values={"body": "It's a test"})
tbl.update(where="id = 6", values={"author": "O'Brien"})

results = tbl.search().where("id >= 1 AND id <= 8").limit(100).to_arrow().to_pylist()
results.sort(key=lambda x: x["id"])

output = []
for r in results:
    output.append({
        "id": r["id"],
        "author": r["author"],
        "body": r["body"]
    })

os.makedirs("/workspace/output", exist_ok=True)
with open("/workspace/output/notes_after.json", "w") as f:
    json.dump(output, f, indent=2)

print("Done")
