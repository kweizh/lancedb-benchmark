import os
import json
import lancedb
import pyarrow as pa

def main():
    lancedb_uri = os.environ.get('LANCEDB_URI', '/workspace/db')
    db = lancedb.connect(lancedb_uri)
    
    schema = pa.schema([
        pa.field("id", pa.int64()),
        pa.field("title", pa.string()),
        pa.field("vector", pa.list_(pa.float32(), 4))
    ])
    
    # Seed the table with rows id = 1..5
    data_seed = [
        {"id": i, "title": f"title-{i}", "vector": [float(i)] * 4}
        for i in range(1, 6)
    ]
    
    table = db.create_table("docs", data=data_seed, schema=schema)
    
    # Save seed version
    seed_version = table.list_versions()[-1]['version']
    
    # 1. Add 3 more rows with id 6, 7, 8.
    data_add = [
        {"id": i, "title": f"title-{i}", "vector": [float(i)] * 4}
        for i in range(6, 9)
    ]
    table.add(data_add)
    
    # 2. update the row where id = 3, setting title to "v3-updated".
    table.update(where="id = 3", values={"title": "v3-updated"})
    
    # 3. delete the row where id = 1.
    table.delete("id = 1")
    
    # Call table.list_versions() and capture the total number of versions.
    all_versions = table.list_versions()
    num_versions = len(all_versions)
    
    # Time-travel back to the early version that exists immediately after the initial seed
    table.checkout(seed_version)
    early_df = table.to_pandas()
    early_version_ids_sorted = sorted(early_df['id'].tolist())
    
    # Call checkout_latest() and read all rows in the latest version.
    table.checkout_latest()
    latest_df = table.to_pandas()
    latest_version_ids_sorted = sorted(latest_df['id'].tolist())
    
    # latest_id3_title
    latest_id3_title = latest_df[latest_df['id'] == 3]['title'].iloc[0]
    
    # Write the final result as JSON to /workspace/output/version_state.json
    output = {
        "num_versions": num_versions,
        "early_version_ids_sorted": early_version_ids_sorted,
        "latest_version_ids_sorted": latest_version_ids_sorted,
        "latest_id3_title": latest_id3_title
    }
    
    os.makedirs("/workspace/output", exist_ok=True)
    with open("/workspace/output/version_state.json", "w") as f:
        json.dump(output, f, indent=2)

if __name__ == "__main__":
    main()
