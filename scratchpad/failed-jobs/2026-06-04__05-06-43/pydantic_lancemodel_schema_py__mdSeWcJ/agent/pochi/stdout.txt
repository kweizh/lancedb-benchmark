import os
import json
import numpy as np
import lancedb
from lancedb.pydantic import LanceModel, Vector

os.makedirs("/workspace/output", exist_ok=True)

db_uri = os.environ.get("LANCEDB_URI", "/workspace/db")
db = lancedb.connect(db_uri)

class Movie(LanceModel):
    id: int
    title: str
    year: int
    genres: list[str]
    summary: str
    vector: Vector(32)

rng = np.random.default_rng(99)
GENRE_LIST = ["drama", "comedy", "action", "sci-fi", "romance"]

movies = []
for i in range(24):
    vec = rng.standard_normal(32).astype('float32').tolist()
    
    title = f"Movie {i:02d}"
    year = 2000 + (i % 25)
    
    genres = [GENRE_LIST[i % 5]]
    if i % 3 == 0:
        genres.append(GENRE_LIST[(i + 2) % 5])
        
    summary = f"Summary for movie {i}."
    
    movie = Movie(
        id=i,
        title=title,
        year=year,
        genres=genres,
        summary=summary,
        vector=vec
    )
    movies.append(movie)

query_vector = rng.standard_normal(32).astype('float32').tolist()

table = db.create_table("movies", schema=Movie, mode="overwrite")
table.add(movies)

results = table.search(query_vector).limit(5).to_list()

output_data = []
top5_ids = []

for r in results:
    output_data.append({
        "id": r["id"],
        "title": r["title"],
        "year": r["year"],
        "_distance": r["_distance"]
    })
    top5_ids.append(r["id"])

with open("/workspace/output/movies_top5.json", "w") as f:
    json.dump(output_data, f, indent=2)

with open("/workspace/output/run.log", "w") as f:
    f.write(f"Top-5 IDs: {json.dumps(top5_ids)}\n")
