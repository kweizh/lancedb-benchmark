schema = pa.schema([
    pa.field('id', pa.int64()),
    pa.field('text', pa.string()),
    pa.field('vector', pa.list_(pa.float32(), 16)),
])

rng = np.random.default_rng(seed=2026)
data = []
for i in range(5):
    data.append({
        'id': i,
        'text': f'doc-{i}',
        'vector': rng.random(16).astype(np.float32).tolist()
    })

tbl = db.create_table('test', data)
print('Version after create:', tbl.version)

# Create tag
tbl.tags.create('v1', tbl.version)
print('Tags:', tbl.tags.list())

# Add more rows
more_data = []
for i in range(5, 8):
    more_data.append({
        'id': i,
        'text': f'doc-{i}',
        'vector': rng.random(16).astype(np.float32).tolist()
    })
tbl.add(more_data)
print('Version after add:', tbl.version)

tbl.tags.create('v2', tbl.version)
print('Tags:', tbl.tags.list())
