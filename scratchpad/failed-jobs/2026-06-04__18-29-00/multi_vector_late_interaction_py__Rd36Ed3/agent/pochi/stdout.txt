# Manual verification
scores = {}
for qi in range(2):
    q_vec = query[qi].ravel()
    res = tbl.search(q_vec).distance_type('cosine').limit(50).to_pandas()
    res['sim'] = 1.0 - res['_distance']
    for doc_id in res['doc_id'].unique():
        doc_rows = res[res['doc_id'] == doc_id]
        max_sim = doc_rows['sim'].max()
        if doc_id not in scores:
            scores[doc_id] = 0.0
        scores[doc_id] += max_sim

# Sort by score descending, tie-break by doc_id ascending
sorted_docs = sorted(scores.items(), key=lambda x: (-x[1], x[0]))
print('Manual top-5:', [d for d, s in sorted_docs[:5]])
