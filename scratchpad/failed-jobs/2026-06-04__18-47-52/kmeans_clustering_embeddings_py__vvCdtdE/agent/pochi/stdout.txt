import lancedb, numpy as np
db = lancedb.connect('lancedb_data')
centroids_table = db.open_table('centroids')

from sklearn.cluster import KMeans
X = np.vstack(db.open_table('embeddings').to_pandas()['vector'].values).astype(np.float64)
kmeans = KMeans(n_clusters=8, random_state=2026, n_init=10).fit(X)

rng = np.random.RandomState(42)
queries = rng.randn(5, 32).astype(np.float64)
for q in queries:
    res = centroids_table.search(q.astype(np.float32)).metric('L2').limit(1).to_pandas()
    p1 = int(res.iloc[0]['cluster_id'])
    p2 = kmeans.predict([q])[0]
    print(p1, p2)
