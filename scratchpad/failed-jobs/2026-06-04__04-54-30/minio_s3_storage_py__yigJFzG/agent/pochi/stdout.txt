```python
import os
import json
import numpy as np
import pyarrow as pa
import lancedb

def main():
    # 1. Read MINIO credentials
    access_key = os.environ.get("MINIO_ACCESS_KEY")
    secret_key = os.environ.get("MINIO_SECRET_KEY")
    if not access_key or not secret_key:
        raise ValueError("MINIO_ACCESS_KEY and MINIO_SECRET_KEY environment variables must be set.")

    # 2. Connect to LanceDB hosted on MinIO
    db_uri = "s3://lance-bucket/"
    storage_options = {
        "endpoint": "http://127.0.0.1:9000",
        "region": "us-east-1",
        "aws_access_key_id": access_key,
        "aws_secret_access_key": secret_key,
        "allow_http": "true"
    }
    
    print(f"Connecting to LanceDB at {db_uri}...")
    db = lancedb.connect(db_uri, storage_options=storage_options)

    # 3. Generate seed data deterministically
    rng = np.random.default_rng(11)
    vectors = rng.standard_normal((16, 8)).astype("float32")

    ids = list(range(16))
    payloads = [f"row-{i:02d}" for i in range(16)]

    # 4. Construct pyarrow Table with schema
    schema = pa.schema([
        pa.field("id", pa.int64()),
        pa.field("payload", pa.string()),
        pa.field("vector", pa.list_(pa.float32(), 8))
    ])

    id_array = pa.array(ids, type=pa.int64())
    payload_array = pa.array(payloads, type=pa.string())
    vector_array = pa.FixedSizeListArray.from_arrays(pa.array(vectors.flatten()), 8)

    table = pa.Table.from_arrays([id_array, payload_array, vector_array], schema=schema)

    # 5. Create table 'vectors_s3'
    table_name = "vectors_s3"
    print(f"Creating table '{table_name}'...")
    tbl = db.create_table(table_name, data=table, mode="overwrite")

    # 6. Generate query vector deterministically using the same generator state
    # Take the next standard_normal(8) sample from a fresh default_rng(11) after consuming 16 seed rows
    rng_query = np.random.default_rng(11)
    _ = rng_query.standard_normal((16, 8)).astype("float32") # consume seed rows
    query = rng_query.standard_normal(8).astype("float32")

    # 7. Run top-3 vector search
    print("Running top-3 vector search...")
    search_results = tbl.search(query).limit(3).to_list()

    # 8. Filter and format results as required: list of {"id": int, "payload": str, "_distance": float}
    output_results = []
    for row in search_results:
        output_results.append({
            "id": int(row["id"]),
            "payload": str(row["payload"]),
            "_distance": float(row["_distance"])
        })

    # 9. Write results to /workspace/output/s3_results.json
    output_dir = "/workspace/output"
    os.makedirs(output_dir, exist_ok=True)
    output_path = os.path.join(output_dir, "s3_results.json")

    print(f"Writing results to {output_path}...")
    with open(output_path, "w") as f:
        json.dump(output_results, f, indent=2)

    print("Done! Results successfully written.")

if __name__ == "__main__":
    main()
```   
