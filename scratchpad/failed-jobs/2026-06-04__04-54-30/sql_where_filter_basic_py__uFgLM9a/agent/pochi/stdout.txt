import lancedb
db = lancedb.connect("/workspace/db")
tbl = db.open_table("events")

def get_filtered_results(predicate):
    res = tbl.search().where(predicate).to_list()
    processed = []
    for item in res:
        processed.append({
            "id": item["id"],
            "event_type": item["event_type"],
            "severity": item["severity"]
        })
    processed.sort(key=lambda x: x["id"])
    return processed

high_severity = get_filtered_results("severity >= 7")
errors_only = get_filtered_results("event_type = '"'"'error'"'"'")
combined = get_filtered_results("event_type = '"'"'warn'"'"' AND severity >= 3")

print("high_severity:")
print(high_severity)
print("\nerrors_only:")
print(errors_only)
print("\ncombined:")
print(combined)
