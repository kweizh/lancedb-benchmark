import lancedb
import pyarrow as pa
import numpy as np

db = lancedb.connect('/tmp/test_db')
schema = pa.schema([
    pa.field('id', pa.int32()),
    pa.field('name', pa.string()),
    pa.field('price', pa.float64()),
    pa.field('tags', pa.list_(pa.string())),
    pa.field('vector', pa.list_(pa.float32(), 4))
])

rng = np.random.default_rng(7)
vectors = [rng.random(4).tolist() for _ in range(6)]
data = [
    {'id': 1, 'name': 'Apple', 'price': 1.20, 'tags': ['fruit', 'red'], 'vector': vectors[0]}
]

tbl = db.create_table('products', data=data, schema=schema, mode='overwrite')
print(tbl.schema)
print(tbl.count_rows())
