# Run 200 random queries and compare cosine vs L2

mismatch_cosine = 0
mismatch_l2 = 0

cents = kmeans.cluster_centers_.astype('float32')

for _ in range(200):
    q = rng.random(32)
    expected = int(kmeans.predict([q])[0])
    
    # Cosine nearest centroid (manual)
    q32 = q.astype('float32')
    q_norm = q32 / (np.linalg.norm(q32) + 1e-9)
    c_norms = cents / (np.linalg.norm(cents, axis=1, keepdims=True) + 1e-9)
    cosine_scores = c_norms @ q_norm
    cosine_pred = int(np.argmax(cosine_scores))
    
    # L2 nearest centroid
    dists = np.linalg.norm(cents - q32, axis=1)
    l2_pred = int(np.argmin(dists))
    
    if cosine_pred != expected:
        mismatch_cosine += 1
    if l2_pred != expected:
        mismatch_l2 += 1
