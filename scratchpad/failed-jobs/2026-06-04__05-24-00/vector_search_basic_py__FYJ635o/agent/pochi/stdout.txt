import numpy as np
rng = np.random.default_rng(42)
vectors = rng.standard_normal((12, 8)).astype(np.float32)
query = rng.standard_normal(8).astype(np.float32)
# Verify top-5 by brute-force L2
dists = [float(np.sum((vectors[i] - query)**2)) for i in range(12)]
ranked = sorted(range(12), key=lambda i: dists[i])
for i in ranked[:5]:
    print(f'id={i}, L2={dists[i]:.6f}')
