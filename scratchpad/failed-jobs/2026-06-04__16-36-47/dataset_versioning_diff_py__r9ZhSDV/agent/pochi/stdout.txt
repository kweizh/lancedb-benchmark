import lancedb
import math

def diff_versions(v_a: int, v_b: int) -> dict:
    db = lancedb.connect("/data/lancedb")
    table = db.open_table("customers")
    
    table.checkout(v_a)
    df_a = table.to_pandas()
    
    table.checkout(v_b)
    df_b = table.to_pandas()
    
    records_a = df_a.to_dict('records')
    records_b = df_b.to_dict('records')
    
    dict_a = {int(r['id']): r for r in records_a}
    dict_b = {int(r['id']): r for r in records_b}
    
    ids_a = set(dict_a.keys())
    ids_b = set(dict_b.keys())
    
    added = list(ids_b - ids_a)
    removed = list(ids_a - ids_b)
    
    modified = []
    for i in ids_a & ids_b:
        ra = dict_a[i]
        rb = dict_b[i]
        
        is_diff = False
        for col in ['name', 'price', 'category']:
            va = ra[col]
            vb = rb[col]
            if col == 'price':
                if not math.isclose(va, vb, rel_tol=1e-5, abs_tol=1e-8):
                    is_diff = True
                    break
            else:
                if va != vb:
                    is_diff = True
                    break
                    
        if is_diff:
            old_vals = {col: ra[col] for col in ['name', 'price', 'category']}
            new_vals = {col: rb[col] for col in ['name', 'price', 'category']}
            modified.append({
                "id": i,
                "old": old_vals,
                "new": new_vals
            })
            
    return {
        "added": added,
        "removed": removed,
        "modified": modified
    }
