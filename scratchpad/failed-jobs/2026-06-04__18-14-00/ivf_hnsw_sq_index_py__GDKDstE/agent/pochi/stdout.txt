import numpy as np
import solution

# Compute brute-force recall test
rng = np.random.default_rng(9999)
queries = rng.standard_normal((50, 128)).astype('float32')
data = solution._vectors  # 1024x128

def cosine_sim(a, b):
    a = a / (np.linalg.norm(a) + 1e-9)
    b = b / (np.linalg.norm(b, axis=1, keepdims=True) + 1e-9)
    return b @ a

recall1_list = []
recall8_list = []
for q in queries:
    sims = cosine_sim(q, data)
    gt_ids = set(np.argsort(-sims)[:10])
    
    r1 = solution.search(q.tolist(), 10, 1)
    r8 = solution.search(q.tolist(), 10, 8)
    
    ids1 = set(row['id'] for row in r1)
    ids8 = set(row['id'] for row in r8)
    
    recall1_list.append(len(gt_ids & ids1) / 10)
